**TypeScript Example**
class Node {
  data: number;
  next: Node | null;

  constructor(data: number) {
    this.data = data;
    this.next = null;
  }
}

let first = new Node(10);
let second = new Node(20);
let third = new Node(30);

first.next = second;
second.next = third;

let current: Node | null = first;
while (current !== null) {
  console.log(current.data);
  current = current.next;
}
